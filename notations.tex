\chapter*{Notations}
\addcontentsline{toc}{chapter}{Notations}

\begin{longtable}{@{} l p{9cm} @{}}
$\mathbb{F}_q$ & Finite field of size $q$ \\[0.4em]
$\mathbb{F}_q^*$ & Multiplicative group of $\mathbb{F}_q$ \\[0.4em]
$\bbz_m$ & residue class of $m$ \\[0.4em]
$\mathbb{N}$ & Set of natural numbers \\[0.4em]
$[n]$ & The set $\{1,2,\dots,n\}$ \\[0.4em]
$\emptyset$ & Empty set \\[0.4em]
$X \sim \mathscr{D}$ & Random variable $X$ having distribution $\mathscr{D}$ \\[0.4em]
$\Delta(\cdot,\cdot)$ & Relative Hamming distance \\[0.4em]
$\E[X]$ & Expectation of random variable $X$ \\[0.4em]
$\D[X]$ & Variance of random variable $X$ \\[0.4em]
$\mathbf{w}(l)$ & $l$-th coordinate of $\mathbf{w}\in\mathbb{F}_q^n$ \\[0.4em]
$D$-evaluation of $h$ over $D$ & Vector of values of $h$ at all points of $D$ \\[0.4em]
$\mathrm{Unif}(S)$ & Uniform distribution over the set $S$ \\[0.4em]
$q$ & A prime power, unless stated otherwise \\[0.4em]
$g^{\mathbf{w}}$ & $(g^{\mathbf{w}(1)}, \ldots, g^{\mathbf{w}(n)})$ \\[0.4em]
$M_{\mathbf{w},\mathbf{v}}$ & $\{g^{\mathbf{w}+\lambda\mathbf{v}} \mid \lambda \in \mathbb{Z}_m\}$ \\[0.4em]
$\mon_{\mathbf{u}}(z_1,\ldots,z_n)$ & $\prod_{\ell} z_\ell^{\mathbf{u}(\ell)}$ in $\mathbb{F}_q[z_1,\dots,z_n]/(z_1^m-1,\dots,z_n^m-1)$ \\[0.4em]
$\langle\mathbf{u},\mathbf{v}\rangle$ & Inner product of vectors $\mathbf{u}$ and $\mathbf{v}$ \\[0.4em]
$\mathrm{supp}(h(y))$ & $\{e\in\mathbb{N}_{\geq0} : y^e\text{ has positive coefficient in }h\}$ \\[0.4em]
$b$-bounded set & Every element of the set ($\subseteq\mathbb{R}$) is less than $b$ \\[0.4em]
$\mathrm{agr}(e,M)$ & Number of agreements of $M$-evaluation of $F$ and $C_m$-evaluation of $h$ \\[0.4em]
$\mathrm{weight}(e,M)$ & $\max\{\mathrm{agr}(e,M) : h\in\mathbb{F}[y],\, h(0)=e,\, \mathrm{supp}(h)\subseteq S\cup\{0\}\}$ \\[0.4em]
$A \tensor B$ & Kronecker product of $A$ and $B$ \\[0.4em]
$A\hprod B$ & Hadamard Product of $A$ and $B$ \\[0.4em]
$\|\bfa\|$ & Hamming weight of $\bfa$\\[0.4em]
\end{longtable}
